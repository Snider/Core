    Options:

    type OpenFileOptions struct {
    Title            string
    DefaultDirectory string
    AllowMultiple    bool
    Filters          []FileFilter
}

type SaveFileOptions struct {
    Title            string
    DefaultDirectory string
    DefaultFilename  string
    Filters          []FileFilter
}

type FileFilter struct {
    DisplayName string
    Pattern     string  // e.g., "*.png;*.jpg"
}
